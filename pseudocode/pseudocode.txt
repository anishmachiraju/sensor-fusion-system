FUNCTIONALITY 0: SimConfig - User Parameter Validation
--------------------------------------------------------------------------------
Purpose:  Hold all user-configurable simulation parameters and validate them.
Input:    User-entered values (speed, turn rate, duration, GPS noise, mass)
Output:   A validated SimConfig struct, or std::invalid_argument exception

STRUCT SimConfig
    speed       (default 5.0,    range 0.1 - 50.0,    m/s)
    turnRate    (default 10.0,   range -90.0 - 90.0,  deg/s)
    duration    (default 0.0,    range 0.0 - 3600.0,  seconds; 0 = infinite)
    gpsNoise    (default 1.5,    range 0.0 - 20.0,    meters)
    mass        (default 1000.0, range 100 - 20000,   kg)

FUNCTION validate() -> void
    IF speed < MIN_SPEED OR speed > MAX_SPEED THEN
        THROW invalid_argument("Speed must be between MIN and MAX")
    IF turnRate < MIN_TURN OR turnRate > MAX_TURN THEN
        THROW invalid_argument("Turn rate must be between MIN and MAX")
    IF duration < MIN_DURATION OR duration > MAX_DURATION THEN
        THROW invalid_argument("Duration must be between MIN and MAX")
    IF gpsNoise < MIN_GPS_NOISE OR gpsNoise > MAX_GPS_NOISE THEN
        THROW invalid_argument("GPS noise must be between MIN and MAX")
    IF mass < MIN_MASS OR mass > MAX_MASS THEN
        THROW invalid_argument("Mass must be between 100 and 20000 kg")


FUNCTIONALITY 1: Generate Sensor Reading (Sensor base class)
--------------------------------------------------------------------------------
Purpose:  Produce a noisy measurement from the true vehicle state.
Input:    True vehicle state (StateEstimate)
Output:   Noisy StateEstimate reading

FUNCTION shouldUpdate(currentTime) -> bool
    interval = 1.0 / updateRate
    IF currentTime - lastUpdateTime >= interval THEN
        RETURN true
    ELSE
        RETURN false

FUNCTION generateReading(trueState) -> StateEstimate
    VALIDATE: trueState.timestamp >= 0
    reading = copy of trueState
    noise = random Gaussian(mean=0, stddev=noiseSigma)
    Add noise + biasOffset to relevant fields of reading
    IF isCalibrated THEN
        reading fields = (reading fields - calibrationOffset) * calibrationScale
    lastReading = reading
    lastUpdateTime = trueState.timestamp
    RETURN reading

FUNCTION calibrate(offset, scale) -> void
    VALIDATE: scale != 0
    calibrationOffset = offset
    calibrationScale = scale
    isCalibrated = true


FUNCTIONALITY 2: GPS Sensor Reading (GPSSensor subclass override)
--------------------------------------------------------------------------------
Purpose:  Produce noisy position data with simulated signal dropout.
Input:    True vehicle state
Output:   Noisy StateEstimate with perturbed positionX, positionY

FUNCTION generateReading(trueState) -> StateEstimate
    VALIDATE: trueState is not empty
    IF random_uniform(0, 1) < dropoutRate THEN
        RETURN lastReading                    // simulate signal loss
    reading = copy of trueState
    noiseX = Gaussian(0, positionNoiseSigma)
    noiseY = Gaussian(0, positionNoiseSigma)
    reading.positionX = reading.positionX + noiseX + positionBias
    reading.positionY = reading.positionY + noiseY + positionBias
    IF isCalibrated THEN
        Apply calibration offset and scale to positionX, positionY
    lastReading = reading
    lastUpdateTime = trueState.timestamp
    RETURN reading


FUNCTIONALITY 3: Weighted-Average Data Fusion (FusionSystem)
--------------------------------------------------------------------------------
Purpose:  Combine calibrated readings from all sensors into one fused estimate.
Input:    Registered sensors with weights
Output:   Fused StateEstimate

FUNCTION addSensor(sensor, weight) -> void
    VALIDATE: sensor is not null, weight > 0
    Append sensor to sensors list
    Append weight to weights list

FUNCTION fuseData() -> StateEstimate
    VALIDATE: sensors list is not empty
    totalWeight = 0
    fused = new StateEstimate with all fields = 0

    FOR i = 0 TO sensors.size() - 1
        reading = sensors[i].getLastReading()
        w = weights[i]
        fused.positionX  = fused.positionX  + w * reading.positionX
        fused.positionY  = fused.positionY  + w * reading.positionY
        fused.velocityX  = fused.velocityX  + w * reading.velocityX
        fused.velocityY  = fused.velocityY  + w * reading.velocityY
        fused.heading    = fused.heading    + w * reading.heading
        fused.temperature= fused.temperature+ w * reading.temperature
        totalWeight = totalWeight + w
    END FOR

    VALIDATE: totalWeight > 0
    Divide every fused field by totalWeight    // normalize
    fusedState = fused
    RETURN fusedState


FUNCTIONALITY 4: Vehicle with Mass-Adjusted Turn Rate
--------------------------------------------------------------------------------
Purpose:  Advance vehicle state via kinematics with mass-affected turning.
Input:    Current state, speed, commandedTurnRate, mass, timestep dt
Output:   Updated StateEstimate

CONSTRUCTOR Vehicle(initState, speed, commandedTurnRate, mass)
    VALIDATE: speed >= 0
    VALIDATE: 100 <= mass <= 20000
    Store fields
    Compute effective turn rate from mass
    Append initState to history

FUNCTION recomputeEffectiveTurnRate() -> void
    factor = (1000 / mass) ^ 0.3
    effectiveTurnRate = commandedTurnRate * factor
    // 100 kg  -> factor 1.99 (turns ~2x faster)
    // 1000 kg -> factor 1.00
    // 20000 kg -> factor 0.41

FUNCTION stepSimulation(dt) -> void
    VALIDATE: dt > 0
    state.heading = state.heading + effectiveTurnRate * dt
    WRAP state.heading into [0, 360)
    headingRad = state.heading * pi / 180
    state.velocityX = speed * cos(headingRad)
    state.velocityY = speed * sin(headingRad)
    state.positionX = state.positionX + state.velocityX * dt
    state.positionY = state.positionY + state.velocityY * dt
    state.timestamp = state.timestamp + dt
    Append state to stateHistory


FUNCTIONALITY 5: Terminal Mode Input Loop
--------------------------------------------------------------------------------
Purpose:  Read configuration from text prompts with validation and re-prompting.
Input:    User typed values via std::cin
Output:   Validated SimConfig

FUNCTION promptDouble(label, default, min, max, units) -> double
    LOOP forever
        Print "  label [default units]: "
        Read line from stdin
        IF EOF or empty THEN RETURN default
        Trim whitespace
        TRY
            v = parse line as double
            IF v < min OR v > max THEN
                Print "  ! out of range, try again"
                CONTINUE
            RETURN v
        CATCH parse error
            Print "  ! not a number, try again"
            CONTINUE
    END LOOP

FUNCTION promptForConfig() -> SimConfig
    Print banner
    cfg.speed     = promptDouble("Vehicle speed", 5.0, 0.1, 50.0, " m/s")
    cfg.turnRate  = promptDouble("Turn rate", 10.0, -90.0, 90.0, " deg/s")
    cfg.duration  = promptDouble("Duration", 30.0, 0.0, 3600.0, " s")
    cfg.gpsNoise  = promptDouble("GPS noise sigma", 1.5, 0.0, 20.0, " m")
    cfg.mass      = promptDouble("Vehicle mass", 1000.0, 100.0, 20000.0, " kg")
    cfg.validate()
    RETURN cfg


FUNCTIONALITY 6: GUI Input Dialog
--------------------------------------------------------------------------------
Purpose:  Modal SFML dialog with five text fields and inline validation.
Input:    Mouse clicks, keyboard text entry
Output:   Validated SimConfig (true) or cancellation (false)

FUNCTION show(outConfig) -> bool
    Build five field structs (label, range, default text)
    Layout field rectangles vertically
    Layout START / CANCEL buttons at bottom
    LOOP while window open
        FOR each event
            IF window closed THEN RETURN false
            IF KeyPressed THEN
                IF Escape THEN RETURN false
                IF Enter  THEN attempt validation
                IF Tab    THEN move focus to next field
                IF Backspace THEN remove last char from focused field
            IF TextEntered THEN
                IF char is digit/dot/minus THEN append to focused field
                Validate the focused field inline
            IF MouseButtonPressed THEN
                Hit-test against fields and buttons
        END FOR
        IF startClicked THEN
            IF validateAll(outConfig) THEN
                close window, RETURN true
            ELSE
                show error, allow user to fix
        Render dialog
    END LOOP
    RETURN false


FUNCTIONALITY 7: Main Simulation Loop (Both Modes)
--------------------------------------------------------------------------------
Purpose:  Orchestrate the full pipeline. Different drivers for terminal vs GUI.

FUNCTION runTerminal() -> int
    cfg = promptForConfig()         // throws if invalid
    sim = makeSimulation(cfg)
    duration = (cfg.duration > 0) ? cfg.duration : 30.0
    WHILE sim.currentTime < duration
        sim.vehicle.stepSimulation(dt)
        FOR each sensor
            IF sensor.shouldUpdate(currentTime) THEN
                sensor.generateReading(trueState)
        fused = sim.fusion.fuseData()
        sim.vehicle.updateState(fused)
        sim.vehicle.displayState()
        currentTime += dt
    END WHILE
    Print summary
    RETURN 0

FUNCTION runGui() -> int
    dialog = new InputDialog
    IF NOT dialog.show(cfg) THEN RETURN 0
    cfg.validate()
    sim = makeSimulation(cfg)
    viz = new Visualizer
    WHILE viz.isOpen()
        viz.handleEvents()
        IF viz.consumeResetRequest() THEN
            sim = makeSimulation(cfg)
            viz.clearForReset()
        IF viz.isRunning() AND elapsed >= frameTime THEN
            sim.vehicle.stepSimulation(dt)
            FOR each sensor
                IF sensor.shouldUpdate(currentTime) THEN
                    sensor.generateReading(trueState)
            fused = sim.fusion.fuseData()
            sim.vehicle.updateState(fused)
            viz.update(state, sensors..., fused)
            currentTime += dt
        END IF
        viz.render()
    END WHILE
    RETURN 0

FUNCTION main(argc, argv) -> int
    Parse command-line arguments (--terminal, --gui, --help)
    TRY
        IF terminalMode THEN RETURN runTerminal()
        ELSE                  RETURN runGui()
    CATCH exception
        Print "[FATAL] " + message
        RETURN 1
